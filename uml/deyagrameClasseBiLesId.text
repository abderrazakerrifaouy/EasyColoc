enum Role {
  USER
  ADMIN
}

' =========================
' CLASSES
' =========================

class User {
  - id : int
  - name : string
  - email : string
  - password : string
  - role : Role
  - reputation : double
}

class Group {
  - id : int
  - name : string
  + generateInvitationToken() : string
}

class Membership {
  - id : int
  - role : string
  - isQuit : boolean
  - user_id : int
  - group_id : int
}

class Category {
  - id : int
  - title : string
  - group_id : int
}

class Invoice {
  - id : int
  - title : string
  - amount : double
  - created_at : date
  - category_id : int
  - membership_id : int
}

class Payment {
  - id : int
  - amount : double
  - payer_membership_id : int
  - receiver_membership_id : int
}

class GroupPayment {
  - id : int
  - payment_id : int
  - invoice_id : int
  - total_amount : double
}
